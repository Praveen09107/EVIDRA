Frontend UI Specification — AIVENTRA Forensic Intelligence Platform
Framework: React 18 + TypeScript
State Management: Zustand (global) + React Query (server state)
Styling: Tailwind CSS v4
Real-time: Native WebSocket (custom hook)
Charts: Recharts + D3 (tower maps)
Routing: React Router v6

3. WebSocket Hook
typescript
// hooks/useCaseWebSocket.ts
export function useCaseWebSocket(caseId: string) {
  const { connect, disconnect, onEvent } = useWsStore();
  const { setStatus, setProgress, setAgentStatus } = usePipelineStore();
  const { setAgentResult } = useCaseStore();
  const token = useAuthToken();

  useEffect(() => {
    connect(caseId, token);

    const unsubscribe = onEvent((event: WsEvent) => {
      switch (event.event) {

        case "pipeline_started":
          setStatus("RUNNING");
          setProgress(0);
          break;

        case "pipeline_progress":
          setProgress(event.payload.progress);
          break;

        case "agent_started":
          setAgentStatus(event.payload.agent_id, "RUNNING");
          break;

        case "agent_complete":
          setAgentStatus(event.payload.agent_id, "COMPLETE");
          // Fetch the result immediately:
          fetchAndStoreAgentResult(event.payload.agent_id, caseId);
          break;

        case "agent_failed":
          setAgentStatus(event.payload.agent_id, "FAILED");
          break;

        case "pipeline_complete":
          setStatus("COMPLETE");
          setProgress(100);
          break;

        case "reasoning_replay_complete":
          // Auto-switch to Replay tab on completion:
          useCaseStore.getState().setActiveTab("replay");
          break;
      }
    });

    return () => {
      unsubscribe();
      disconnect();
    };
  }, [caseId, token]);
}

const AGENT_ORDER = [

const STATUS_COLORS: Record<AgentTaskStatus, string> = {
  WAITING:    "bg-[--color-surface-dynamic]",
  DISPATCHED: "bg-[--color-gold]",
  RUNNING:    "bg-[--color-primary] animate-pulse",
  COMPLETE:   "bg-[--color-success]",
  FAILED:     "bg-[--color-error]",
  SKIPPED:    "bg-[--color-border]"

export function PipelineStatusBar({ status, progress, agentStatuses }) {
  if (status === "PENDING") return null;

  return (
    <div className="px-6 py-3 bg-[--color-surface] border-b border-[--color-border]">
      {/* Progress bar */}
      <div className="flex items-center gap-3 mb-2">
        <span className="text-[--text-sm] text-[--color-text-muted]">
          {status === "RUNNING" ? "Analyzing..." : status}
        </span>
        <div className="flex-1 h-1.5 bg-[--color-surface-dynamic] rounded-full">
          <div
            className="h-full bg-[--color-primary] rounded-full transition-all duration-500"
            style={{ width: `${progress}%` }}
          />
        </div>
        <span className="text-[--text-sm] font-medium tabular-nums">
          {progress}%
        </span>
      </div>

      {/* Per-agent dots */}
      <div className="flex gap-2 flex-wrap">
        {AGENT_ORDER.map(agent => (
          <div key={agent} className="flex items-center gap-1">
            <div className={`w-2 h-2 rounded-full ${
              STATUS_COLORS[agentStatuses[agent] ?? "WAITING"]
            }`} />
            <span className="text-[--text-xs] text-[--color-text-muted]">
              {agent.replace(/_/g, " ")}
            </span>
          </div>
        ))}
      </div>
    </div>
  );
}

export function ReplayTab() {
  const { replayReport } = useCaseStore();

7.1 Replay Step Table

// features/replay/ReplayStepTable.tsx

const CONFIDENCE_ROW_COLOR = (conf: number) =>
  conf >= 0.80 ? "border-l-[--color-success]"
  : conf >= 0.60 ? "border-l-[--color-gold]"
  : "border-l-[--color-error]";

export function ReplayStepTable({ steps, gaps, filter }) {
  const filtered = filter === "ALL"
    ? steps
    : steps.filter(s => s.agent_id === filter);

  const gapStepIds = new Set(gaps.map(g => g.step_id));

  return (
    <div className="rounded-lg border border-[--color-border] overflow-hidden">
      <table className="w-full text-[--text-sm]">
        <thead className="bg-[--color-surface-offset]">
          <tr>
            <th className="px-3 py-2 text-left font-medium w-12">#</th>
            <th className="px-3 py-2 text-left font-medium w-36">Agent</th>
            <th className="px-3 py-2 text-left font-medium w-40">Trigger</th>
            <th className="px-3 py-2 text-left font-medium">Input</th>
            <th className="px-3 py-2 text-left font-medium">Conclusion</th>
            <th className="px-3 py-2 text-left font-medium w-20">Conf.</th>
            <th className="px-3 py-2 text-left font-medium w-20">ms</th>
          </tr>
        </thead>
        <tbody>
          {filtered.map((step, i) => (
            <>
              {/* Gap warning inline */}
              {gapStepIds.has(step.step_id) && (
                <tr key={`gap-${step.step_id}`}>
                  <td colSpan={7} className="px-3 py-1.5
                      bg-[--color-warning-highlight] text-[--color-warning]
                      text-[--text-xs]">
                    ⚠ Reasoning gap detected before step {step.global_index}
                  </td>
                </tr>
              )}

              <tr key={step.step_id}
                  className={`border-l-2 ${CONFIDENCE_ROW_COLOR(step.confidence)}
                    hover:bg-[--color-surface-offset] transition-colors
                    ${i % 2 === 0 ? "bg-[--color-bg]" : "bg-[--color-surface]"}`}>
                <td className="px-3 py-2 tabular-nums text-[--color-text-muted]">
                  {step.global_index}
                </td>
                <td className="px-3 py-2">
                  <AgentBadge agentId={step.agent_id} />
                </td>
                <td className="px-3 py-2">
                  <TriggerTypeBadge type={step.trigger_type} />
                </td>
                <td className="px-3 py-2 text-[--color-text-muted] max-w-xs truncate"
                    title={step.input_summary}>
                  {step.input_summary}
                </td>
                <td className="px-3 py-2 max-w-sm">
                  {step.interpretation}
                </td>
                <td className="px-3 py-2 tabular-nums font-medium"
                    style={{ color: confidenceColor(step.confidence) }}>
                  {(step.confidence * 100).toFixed(0)}%
                </td>
                <td className="px-3 py-2 tabular-nums text-[--color-text-muted]">
                  {step.duration_ms ?? "–"}
                </td>
              </tr>

              {/* Warnings under row */}
              {step.warnings.length > 0 && (
                <tr key={`warn-${step.step_id}`}
                    className="bg-[--color-warning-highlight]">
                  <td colSpan={7} className="px-3 py-1 text-[--text-xs]
                      text-[--color-warning]">
                    {step.warnings.join(" · ")}
                  </td>
                </tr>
              )}
            </>
          ))}
        </tbody>
      </table>
    </div>
  );
}

            {hypothesisResult.weakening_evidence.map(e => (
              <li key={e} className="flex items-start gap-2 text-[--text-sm]">
                <XCircle size={14} className="text-[--color-error] mt-0.5 shrink-0" />
                {e}
              </li>
            ))}
          </ul>
        </SectionCard>
      </div>

      {/* Contradictions */}
      {hypothesisResult.contradictions.length > 0 && (
        <div className="col-span-12">
          <ContradictionBanner contradictions={hypothesisResult.contradictions} />
        </div>
      )}

      {/* Recommended actions */}
      <div className="col-span-12">
        <SectionCard title="Recommended Next Steps">
          <ol className="space-y-2">
            {hypothesisResult.recommended_actions.map((a, i) => (
              <li key={i} className="flex items-start gap-3 text-[--text-sm]">
                <span className="w-6 h-6 rounded-full bg-[--color-primary-highlight]
                                  text-[--color-primary] text-[--text-xs] font-bold
                                  flex items-center justify-center shrink-0">
                  {i + 1}
                </span>
                {a}
              </li>
            ))}
          </ol>
        </SectionCard>
      </div>

    </div>
  );
}

// api/client.ts
import axios from "axios";
import { useAuthStore } from "@/store/authStore";

export const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 30_000
});

// Attach token:
apiClient.interceptors.request.use(config => {
  const token = useAuthStore.getState().accessToken;
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Handle 401 → refresh:
apiClient.interceptors.response.use(
  r => r,
  async error => {
    if (error.response?.status === 401) {
      await useAuthStore.getState().refreshToken();
      return apiClient.request(error.config);
    }
    return Promise.reject(error);
  }
);

// Typed API calls:
export const casesApi = {
  list:    ()           => apiClient.get<Case[]>("/cases"),

export const filesApi = {

    });
  }

AIVENTRA — Layer 3: Agent Orchestration & Forensic Intelligence Layer
Full Technical Specification
Layer: 3 of 4 | Domain: Multi-Agent Pipeline, Orchestrator, All 11 Forensic Agents, Inter-Agent Communication, mTLS, Circuit Breakers
Runtime: Python 3.12 (FastAPI per agent) | Ports: 8010–8021 (one per agent) + 8009 (Orchestrator)

PART I — LAYER PURPOSE & BOUNDARIES
Layer 3 is the forensic intelligence engine of AIVENTRA. It takes encrypted evidence references from Layer 2, orchestrates 11 specialized agents in a dependency-aware DAG, and produces signed structured outputs that feed Layer 4's report and replay systems.

Five core responsibilities:

Orchestration — DAG-aware pipeline dispatcher with retry, circuit breaking, and partial failure recovery

Evidence Decryption Bridge — secure internal evidence reading via task-token-gated calls to Layer 2

Forensic Analysis — 11 specialized agents across 3 execution tiers covering all forensic domains

Inter-Agent Security — mTLS mutual authentication, one-time task tokens, output signing

Replay Chain Construction — every inference step logged as a cryptographically linked step record

Boundary contracts:

Receives: pipeline_run_id, case_id, evidence_ids[] from Layer 1 API (via Redis Stream)

Reads: encrypted evidence via evidence_reader service (Layer 2 internal, not HTTP)

Writes: agent_runs, agent_outputs, timeline_events, hypothesis_snapshots, replay_steps to PostgreSQL

Feeds: Layer 4 (Report Generator, Reasoning Replay Viewer) reads all agent_outputs and replay_steps

Emits: WebSocket events for every state transition to Layer 1 WS manager

2.3 Data Models — Orchestrator Owned
sql
-- agent_runs: one row per agent per pipeline execution
CREATE TABLE agent_runs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    pipeline_run_id     UUID NOT NULL,
    case_id             UUID NOT NULL REFERENCES cases(id),
    agent_id            TEXT NOT NULL,
    tier                INTEGER NOT NULL,
    status              TEXT NOT NULL DEFAULT 'PENDING' CHECK (status IN (
                          'PENDING','RUNNING','DONE','FAILED','SKIPPED'
                        )),
    retry_count         INTEGER NOT NULL DEFAULT 0,
    started_at          TIMESTAMPTZ,
    completed_at        TIMESTAMPTZ,
    duration_ms         INTEGER,
    llm_tokens_used     INTEGER DEFAULT 0,
    error_message       TEXT,
    task_token_jti      TEXT,                -- JTI of the task token used
    archived            BOOLEAN NOT NULL DEFAULT FALSE,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_runs_pipeline     ON agent_runs(pipeline_run_id);
CREATE INDEX idx_runs_case_status  ON agent_runs(case_id, status);
CREATE INDEX idx_runs_agent        ON agent_runs(agent_id, pipeline_run_id);

-- agent_outputs: signed output blobs from each agent
CREATE TABLE agent_outputs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    run_id              UUID NOT NULL REFERENCES agent_runs(id),
    pipeline_run_id     UUID NOT NULL,
    case_id             UUID NOT NULL,
    agent_id            TEXT NOT NULL,
    output_schema       TEXT NOT NULL,       -- e.g. "AutopsyOutput_v1"
    output_json         JSONB NOT NULL,      -- full signed output payload
    output_hash         TEXT NOT NULL,       -- SHA-256 of canonical payload JSON
    signature           TEXT NOT NULL,       -- RSA-PKCS1v15 sig of output_hash
    model_version       TEXT NOT NULL,
    superseded          BOOLEAN NOT NULL DEFAULT FALSE,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- immutable after insert
CREATE TRIGGER trg_prevent_output_update
  BEFORE UPDATE ON agent_outputs
  FOR EACH ROW
  WHEN (OLD.superseded = FALSE)             -- allow superseded flag update only
  EXECUTE FUNCTION prevent_modification();

CREATE INDEX idx_outputs_case_agent    ON agent_outputs(case_id, agent_id);
CREATE INDEX idx_outputs_pipeline      ON agent_outputs(pipeline_run_id);
CREATE INDEX idx_outputs_latest        ON agent_outputs(case_id, agent_id, superseded)
             WHERE superseded = FALSE;

-- replay_steps: cryptographic inference chain
CREATE TABLE replay_steps (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    run_id              UUID NOT NULL REFERENCES agent_runs(id),
    pipeline_run_id     UUID NOT NULL,
    case_id             UUID NOT NULL,
    seq                 INTEGER NOT NULL,
    agent_id            TEXT NOT NULL,
    trigger_type        TEXT NOT NULL CHECK (trigger_type IN (
                          'INPUT','RULE','MODEL_OUTPUT','BAYESIAN_UPDATE',
                          'LLM_EXTRACTION','LLM_NARRATIVE','CONSISTENCY_CHECK',
                          'INTEGRITY_GATE','ANOMALY_DETECTED','HOTSPOT_IDENTIFIED'
                        )),
    input_summary       TEXT NOT NULL,
    interpretation      TEXT NOT NULL,       -- PII-masked narrative
    confidence          NUMERIC(5,4),        -- 0.0000 to 1.0000
    evidence_ids        UUID[],
    input_data          JSONB,               -- full input snapshot (masked)
    model_version       TEXT,
    step_hash           TEXT NOT NULL,       -- SHA-256 of canonical step content
    prev_step_hash      TEXT NOT NULL,       -- previous step's step_hash (chain linkage)
    signature           TEXT NOT NULL,       -- RS256 sig of step_hash
    timestamp           TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_replay_run_seq    ON replay_steps(run_id, seq);
CREATE INDEX idx_replay_case       ON replay_steps(case_id, pipeline_run_id);
CREATE INDEX idx_replay_ordered    ON replay_steps(pipeline_run_id, seq ASC);

-- timeline_events: digital forensic events
CREATE TABLE timeline_events (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    case_id             UUID NOT NULL REFERENCES cases(id),
    pipeline_run_id     UUID NOT NULL,
    event_time          TIMESTAMPTZ NOT NULL,
    event_time_end      TIMESTAMPTZ,         -- for duration events
    event_type          TEXT NOT NULL,       -- CALL, SMS, LOCATION, TRANSACTION, etc.
    source              TEXT NOT NULL,       -- CDR, LOCATION_DATA, FINANCIAL_RECORDS, etc.
    actor_masked        TEXT,                -- [PERSON_1]
    counterparty_masked TEXT,                -- [PERSON_2]
    location_masked     TEXT,                -- [ADDRESS_1]
    event_details       JSONB,              -- full masked event data
    is_anomalous        BOOLEAN NOT NULL DEFAULT FALSE,
    anomaly_score       NUMERIC(8,6),
    anomaly_flags       TEXT[],
    hotspot_id          UUID,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_timeline_case_time   ON timeline_events(case_id, event_time);
CREATE INDEX idx_timeline_anomalous   ON timeline_events(case_id, is_anomalous)
             WHERE is_anomalous = TRUE;
CREATE INDEX idx_timeline_case_range  ON timeline_events(case_id, event_time, event_time_end);
CREATE INDEX idx_timeline_source      ON timeline_events(case_id, source);

═══════════════════════════════════════════════════════════════
STEP 1: API Enqueue (Layer 1 handler: POST /agents/run)
═══════════════════════════════════════════════════════════════

1.1  After Layer 1 auth + RBAC:
     require_role(["ADMIN", "LEAD_INVESTIGATOR"])
     await require_case_access(user, case_id, db)

1.2  Validate evidence_ids all belong to case_id:
     SELECT COUNT(*) FROM evidence_files
     WHERE id = ANY($evidence_ids)
     AND case_id = $case_id
     AND status = 'ACTIVE'

1.3  Check no active pipeline already running:
     SELECT COUNT(*) FROM agent_runs
     WHERE case_id=$1
     AND pipeline_run_id = (latest)
     AND status = 'RUNNING'

1.4  Generate pipeline_run_id = uuid4()

1.5  Create PENDING agent_run rows (one per agent, 11 total):
     async with db.transaction():

     await redis.xadd(

     )

1.7  Audit: PIPELINE_TRIGGERED
1.8  WS broadcast: PIPELINE_TRIGGERED → {pipeline_run_id, total_agents: 11}

1.9  HTTP 202 Accepted {
       "pipeline_run_id": str(pipeline_run_id),
       "status":          "QUEUED",
       "estimated_duration_minutes": 8
     }

STEP 2: Orchestrator Picks Up Task
═══════════════════════════════════════════════════════════════

2.1  Orchestrator main loop (runs continuously):
     while True:
       messages = await redis.xreadgroup(
           groupname="orchestrators",
           consumername=f"orchestrator-{hostname}",
           streams={"aiventra:pipeline:tasks": ">"},
           count=1,
           block=5000   ← block 5 seconds waiting for messages
       )
       if messages:
         stream_name, entries = messages[0]
         for msg_id, fields in entries:
           await _process_pipeline_task(fields, msg_id)
           await redis.xack("aiventra:pipeline:tasks", "orchestrators", msg_id)

         pipeline_run_id = UUID(fields["pipeline_run_id"]),

         triggered_by    = UUID(fields["triggered_by"]),
         start_time      = datetime.utcnow(),

         agent_outputs   = {},
         replay_steps    = [],
         replay_seq      = 0

2.3  UPDATE agent_runs SET status='PENDING' WHERE pipeline_run_id=$1
     (already set in step 1.5, this is a sanity reset)

2.4  WS broadcast: PIPELINE_STARTED → {pipeline_run_id, queue_wait_ms}
3.2 Tier-by-Tier DAG Execution
text
═══════════════════════════════════════════════════════════════
TIER 1 EXECUTION: evidence_parser → ocr → format_normalizer
═══════════════════════════════════════════════════════════════

NOTE: Within Tier 1, ocr depends on evidence_parser output,
      and format_normalizer depends on both. So Tier 1 is
      actually sequential with internal dependency, not fully parallel.
      True parallel = tiers 2 and 3.

STEP 3.1: Dispatch evidence_parser
────────────────────────────────────────────────────────────────
3.1.1  Circuit breaker check:
       cb = circuit_breakers["evidence_parser"]
       if cb.state == "OPEN":
         if time.time() < cb.open_until:
           → SKIP this agent, mark SKIPPED
           → Pipeline cannot proceed (evidence_parser is CRITICAL)
           → Fail entire pipeline with CRITICAL_AGENT_UNAVAILABLE

3.1.2  Issue task token:
       task_token = issue_task_token(
           target="evidence_parser",
           case_id=ctx.case_id,
           operation="run",
           ttl=300
       )
       Store jti → UPDATE agent_runs SET task_token_jti=jti WHERE ...

3.1.3  Build request payload:
       payload = {
           "pipeline_run_id": str(ctx.pipeline_run_id),
           "case_id":         str(ctx.case_id),
           "evidence_ids":    [str(e) for e in ctx.evidence_ids],
           "task_token":      task_token,
           "config":          ctx.config.dict()
       }

3.1.4  Mark RUNNING:
       UPDATE agent_runs SET status='RUNNING', started_at=NOW()
       WHERE pipeline_run_id=$1 AND agent_id='evidence_parser'
       WS broadcast: AGENT_STARTED → {agent_id: evidence_parser}

3.1.5  Dispatch with mTLS:
       async with httpx.AsyncClient(
           cert=("certs/orchestrator.crt", "keys/orchestrator.key"),
           verify="certs/internal_ca.crt",
           timeout=AGENT_DAG["evidence_parser"]["timeout_sec"]
       ) as client:
           response = await client.post(
               "https://evidence_parser:8020/run",
               json=payload
           )

3.1.6  Handle response:
       if response.status_code == 200:
         output = response.json()
         → verify output signature (Section 5.1)
         → store in agent_outputs table
         → ctx.agent_outputs["evidence_parser"] = output
         → ctx.agent_statuses["evidence_parser"] = "DONE"
         circuit_breakers["evidence_parser"].record_success()
         UPDATE agent_runs SET status='DONE', completed_at=NOW(),
           duration_ms=elapsed, llm_tokens_used=output.meta.llm_tokens
         WS broadcast: AGENT_COMPLETE
       else:
         → handle failure (Section 3.3)

STEP 3.2: Dispatch ocr (depends on evidence_parser DONE)
────────────────────────────────────────────────────────────────
3.2.1  Check dependency: evidence_parser must be DONE
       If evidence_parser FAILED:
         → check if any evidence files need OCR
         → evidence_parser output has routing: needs_ocr: bool
         → If routing unavailable: skip OCR, forward_normalizer uses raw Layer 2 text
         → Mark ocr as SKIPPED (not FAILED — graceful)

3.2.2  Routing decision from evidence_parser output:
       needs_ocr_ids = [f for f in evidence_parser_output.files
                        if f.needs_ocr == True]
       If len(needs_ocr_ids) == 0:
         → Mark ocr SKIPPED (no scanned documents in this case)
         → Log replay step: {trigger: INPUT, interpretation: "No scanned files requiring OCR"}
         → Skip dispatch
       Else:
         → Dispatch ocr with needs_ocr_ids

3.2.3  Same dispatch pattern (steps 3.1.1–3.1.6) for ocr

STEP 3.3: Dispatch format_normalizer (depends on evidence_parser + ocr DONE/SKIPPED)
────────────────────────────────────────────────────────────────
3.3.1  Wait for both dependencies in DONE or SKIPPED state
3.3.2  Assemble input: combine evidence_parser text extracts + ocr results
3.3.3  Dispatch format_normalizer with combined text corpus

═══════════════════════════════════════════════════════════════
TIER 2 EXECUTION: autopsy + tod + digital_timeline + anomaly_detector
═══════════════════════════════════════════════════════════════

NOTE: autopsy, tod, digital_timeline run fully in PARALLEL.
      anomaly_detector depends on digital_timeline.

STEP 3.4: Dispatch Tier 2 parallel group
────────────────────────────────────────────────────────────────
3.4.1  Confirm format_normalizer is DONE
3.4.2  normalized_output = ctx.agent_outputs["format_normalizer"]

3.4.3  Dispatch autopsy, tod, digital_timeline CONCURRENTLY:
       tasks = [
           _dispatch_agent("autopsy",          ctx, depends_output=normalized_output),
           _dispatch_agent("tod",               ctx, depends_output=normalized_output),
           _dispatch_agent("digital_timeline",  ctx, depends_output=normalized_output),
       ]
       results = await asyncio.gather(*tasks, return_exceptions=True)

3.4.4  Process results:
       for agent_id, result in zip(["autopsy","tod","digital_timeline"], results):
         if isinstance(result, Exception):
           await _handle_agent_failure(agent_id, ctx, str(result))
         else:
           ctx.agent_outputs[agent_id] = result
           ctx.agent_statuses[agent_id] = "DONE"

3.4.5  Wait for digital_timeline DONE/FAILED before anomaly_detector:
       if ctx.agent_statuses["digital_timeline"] == "DONE":
         await _dispatch_agent("anomaly_detector", ctx,
                                depends_output=ctx.agent_outputs["digital_timeline"])
       else:
         await _mark_agent_skipped("anomaly_detector",
                                    reason="digital_timeline failed")

═══════════════════════════════════════════════════════════════
TIER 3 EXECUTION: hotspot + hypothesis + replay + report (parallel with deps)
═══════════════════════════════════════════════════════════════

STEP 3.5: Dispatch Tier 3
────────────────────────────────────────────────────────────────
3.5.1  hotspot depends on: anomaly_detector + tod
       hypothesis depends on: autopsy + tod + hotspot + digital_timeline
       reasoning_replay depends on: hypothesis + hotspot + anomaly_detector
       report_generator depends on: ALL tier 2 + reasoning_replay + hypothesis

3.5.2  Dispatch hotspot (after anomaly_detector + tod done):
       await _dispatch_agent("hotspot_correlation", ctx, depends_outputs={
           "anomaly_detector": ctx.agent_outputs.get("anomaly_detector"),
           "tod":              ctx.agent_outputs.get("tod")
       })

3.5.3  Dispatch hypothesis_manager (after autopsy + tod + hotspot + digital_timeline):
       await _dispatch_agent("hypothesis_manager", ctx, depends_outputs={
           "autopsy":             ctx.agent_outputs.get("autopsy"),
           "tod":                 ctx.agent_outputs.get("tod"),
           "hotspot_correlation": ctx.agent_outputs.get("hotspot_correlation"),
           "digital_timeline":    ctx.agent_outputs.get("digital_timeline")
       })
       WS broadcast on completion: HYPOTHESIS_UPDATED

3.5.4  Dispatch reasoning_replay:
       await _dispatch_agent("reasoning_replay", ctx, depends_outputs={
           "hypothesis_manager":   ctx.agent_outputs.get("hypothesis_manager"),
           "hotspot_correlation":  ctx.agent_outputs.get("hotspot_correlation"),
           "anomaly_detector":     ctx.agent_outputs.get("anomaly_detector"),

3.5.5  Dispatch report_generator (after ALL others):
       await _dispatch_agent("report_generator", ctx, depends_outputs={
           a: ctx.agent_outputs.get(a) for a in AGENT_DAG
           if a != "report_generator"
       })

═══════════════════════════════════════════════════════════════
STEP 3.6: Pipeline Completion
═══════════════════════════════════════════════════════════════

3.6.1  Tally final statuses:
       failed = [a for a in AGENT_DAG if ctx.agent_statuses[a] == "FAILED"]

async def _handle_agent_failure(agent_id, ctx, error_msg):
═══════════════════════════════════════════════════════════════

1.  Load current retry_count from DB:
    run = await db.fetchrow(
        "SELECT retry_count FROM agent_runs
         WHERE pipeline_run_id=$1 AND agent_id=$2",
        ctx.pipeline_run_id, agent_id
    )
    retry_count = run.retry_count

2.  Circuit breaker: record failure:
    circuit_breakers[agent_id].record_failure()
    → failure_count += 1
    → If failure_count >= FAILURE_THRESHOLD (3):
      state = "OPEN", open_until = time.time() + RECOVERY_TIMEOUT (60s)

3.  If retry_count < MAX_RETRIES (3):
    a.  Compute delay: delay = (2 ** retry_count) * 5 + random.uniform(0, 1)
        → retry_count=0: ~5s | retry_count=1: ~10s | retry_count=2: ~20s
    b.  WS broadcast: AGENT_RETRYING → {agent_id, retry_count, delay}
    c.  UPDATE agent_runs SET retry_count=retry_count+1, status='PENDING'
    d.  await asyncio.sleep(delay)
    e.  Re-dispatch: return await _dispatch_agent(agent_id, ctx, ...)

4.  If retry_count >= MAX_RETRIES:
    a.  UPDATE agent_runs SET status='FAILED', error_message=$error_msg,
                              completed_at=NOW()
    b.  ctx.agent_statuses[agent_id] = "FAILED"
    c.  Audit: AGENT_RUN_FAILURE → {agent_id, error_msg, retry_count}
    d.  WS broadcast: AGENT_ERROR → {agent_id, error_summary, retry_count}

5.  Downstream impact analysis:
    For each downstream agent that depends on agent_id:
      if ALL dependencies failed (no partial output available):
        → Mark downstream SKIPPED (not FAILED — dependency unavailable)
      else:
        → Dispatch downstream with available partial outputs
        → Downstream agents must handle missing inputs gracefully

CIRCUIT BREAKER STATES:
──────────────────────────────────────────────────────────────
CLOSED:
  - failure_count < FAILURE_THRESHOLD
  - All requests pass through normally
  - On each failure: failure_count++
  - On each success: failure_count = 0

OPEN:
  - failure_count >= FAILURE_THRESHOLD (3)
  - Requests REJECTED immediately (no dispatch)
  - open_until = time.time() + RECOVERY_TIMEOUT (60s)
  - After RECOVERY_TIMEOUT: transition to HALF_OPEN

HALF_OPEN:
  - One probe request allowed
  - If succeeds: → CLOSED, failure_count = 0
  - If fails: → OPEN, reset timer

class CircuitBreaker:
    state: str = "CLOSED"
    failure_count: int = 0
    FAILURE_THRESHOLD: int = 3
    RECOVERY_TIMEOUT: int = 60
    open_until: float = 0

    def record_success(self):
        self.failure_count = 0
        self.state = "CLOSED"

    def record_failure(self):
        self.failure_count += 1
        if self.failure_count >= self.FAILURE_THRESHOLD:
            self.state = "OPEN"
            self.open_until = time.time() + self.RECOVERY_TIMEOUT

    def can_attempt(self) -> bool:
        if self.state == "CLOSED":   return True
        if self.state == "OPEN":
            if time.time() > self.open_until:
                self.state = "HALF_OPEN"
                return True
            return False
        if self.state == "HALF_OPEN": return True  # one probe
        return False

       max_pmi_hours = (discovery_time - last_seen_time).total_seconds() / 3600

     else:
       prior = uniform_prior(0, 96)   ← wide prior if no timeline constraints

       "physics": 0.4 if pmi_physics else 0.0,
       "rigor":   0.2 if rigor_stage  else 0.0,
       "livor":   0.15 if livor_stage else 0.0,
       "decomp":  0.1 if decomp_stage else 0.0,
       "ml":      0.35 if pmi_ml_mean else 0.0
     }
     # Normalize weights
     total_w = sum(component_weights.values())
     component_weights = {k: v/total_w for k,v in component_weights.items()}

     L_physics = gaussian_pdf(pmi_grid, pmi_physics.mean_hours,
                               (pmi_physics.upper95-pmi_physics.lower95)/4) if pmi_physics else 1.0

     L_combined = (
         L_physics ** component_weights["physics"] *
         L_rigor   ** component_weights["rigor"]   *
         L_livor   ** component_weights["livor"]   *
         L_decomp  ** component_weights["decomp"]  *
         L_ml      ** component_weights["ml"]
     )

6.3  Posterior = Prior * L_combined (unnormalized)
     posterior = prior * L_combined
     posterior = posterior / posterior.sum()            ← normalize to sum=1

6.4  Monte Carlo sampling (10,000 samples):
     samples = np.random.choice(pmi_grid, size=10000, p=posterior)

6.5  Extract statistics:

     pmi_mean    = np.mean(samples)
     pmi_lower68 = np.percentile(samples, 16)
     pmi_upper68 = np.percentile(samples, 84)
     pmi_lower95 = np.percentile(samples, 2.5)
     pmi_upper95 = np.percentile(samples, 97.5)

6.6  Convert PMI to TOD (subtract from discovery):

     tod_lower95 = discovery_time - timedelta(hours=pmi_upper95)
     tod_upper95 = discovery_time - timedelta(hours=pmi_lower95)

5.2 Task Token System (One-Time Use JWT)
python
TASK_TOKEN_SPEC:
──────────────────────────────────────────────────────────────
Algorithm:  RS256 (RSA + SHA-256)
Signing Key: TASK_TOKEN_PRIVATE_KEY (RSA-2048, from env/secrets)
Verify Key:  TASK_TOKEN_PUBLIC_KEY  (shared to all agents at startup)

CLAIMS:
{
  "iss": "aiventra-orchestrator",
  "sub": "evidence_parser",          ← target agent
  "op":  "run",                      ← permitted operation
  "cid": "case-uuid-here",           ← case_id restriction
  "pid": "pipeline-uuid-here",       ← pipeline_run_id restriction
  "jti": "unique-token-id-uuid4",    ← one-time use ID
  "iat": 1746732000,
  "exp": 1746732300                  ← TTL: 300 seconds (5 minutes)
}

ISSUE (orchestrator):
──────────────────────────────────────────────────────────────
def issue_task_token(target_agent, case_id, pipeline_run_id, ttl=300):
    jti = str(uuid4())
    now = int(time.time())
    payload = {
        "iss": "aiventra-orchestrator",
        "sub": target_agent,
        "op":  "run",
        "cid": str(case_id),
        "pid": str(pipeline_run_id),
        "jti": jti,
        "iat": now,
        "exp": now + ttl
    }
    token = jwt.encode(payload, TASK_TOKEN_PRIVATE_KEY, algorithm="RS256")
    # Mark JTI as "issued" in Redis (for replay detection):
    await redis.setex(f"task_token:issued:{jti}", ttl + 60, "1")
    return token

VERIFY (agent, on every /run request):
──────────────────────────────────────────────────────────────
async def verify_task_token(token, expected_agent, case_id, pipeline_run_id):
    # 1. Decode + verify signature + expiry:
    payload = jwt.decode(
        token,
        TASK_TOKEN_PUBLIC_KEY,
        algorithms=["RS256"],
        options={"verify_exp": True}
    )

    # 2. Validate claims:
    assert payload["iss"] == "aiventra-orchestrator",  raise InvalidToken
    assert payload["sub"] == expected_agent,           raise WrongAgent
    assert payload["op"]  == "run",                    raise InvalidOp
    assert payload["cid"] == str(case_id),             raise CaseMismatch
    assert payload["pid"] == str(pipeline_run_id),     raise PipelineMismatch

    # 3. One-time use check (replay prevention):
    jti = payload["jti"]
    already_used = await redis.exists(f"task_token:used:{jti}")
    if already_used:
        raise TokenAlreadyUsed(jti)

    # 4. Mark as used (atomic — race condition safe):
    result = await redis.set(
        f"task_token:used:{jti}",
        "1",
        ex=600,          ← keep for 10 min after use (for audit)
        nx=True          ← only set if NOT exists (atomic one-time use)
    )
    if not result:
        raise TokenAlreadyUsed(jti)   ← concurrent replay attempt

    return payload
5.3 Agent Output Signing & Verification
python
OUTPUT_SIGNING_SPEC:
──────────────────────────────────────────────────────────────
Algorithm:  RSA-PKCS1v15 + SHA-256
Key source: AGENT_PRIVATE_KEY (per-agent, from env or secrets mount)
            Orchestrator uses ORCHESTRATOR_PRIVATE_KEY for replay chain

SIGN (each agent, before returning output):
──────────────────────────────────────────────────────────────
def sign_agent_output(output_dict, agent_id, case_id, run_id):
    # Build canonical JSON (deterministic key ordering, no whitespace):
    canonical = json.dumps(output_dict, sort_keys=True, separators=(",",":"))

    # Compute content hash:
    output_hash = sha256(canonical.encode()).hexdigest()

    # Sign the hash:
    from Crypto.Signature import pkcs1_15
    from Crypto.Hash import SHA256
    from Crypto.PublicKey import RSA

    key = RSA.import_key(AGENT_PRIVATE_KEY)
    h   = SHA256.new(output_hash.encode())
    sig = pkcs1_15.new(key).sign(h)
    signature = base64.b64encode(sig).decode()

    return {
        "agent_id":       agent_id,
        "case_id":        str(case_id),
        "run_id":         str(run_id),
        "schema_version": "1.0",
        "output_hash":    output_hash,
        "signature":      signature,
        "signed_at":      datetime.utcnow().isoformat(),
        "payload":        output_dict
    }

VERIFY (orchestrator, on receiving each agent response):
──────────────────────────────────────────────────────────────
def verify_agent_output(signed_output, expected_agent_id):
    # 1. Check agent_id matches expectation:
    assert signed_output["agent_id"] == expected_agent_id

    # 2. Recompute canonical hash:
    canonical  = json.dumps(signed_output["payload"],
                             sort_keys=True, separators=(",",":"))
    recomputed = sha256(canonical.encode()).hexdigest()
    assert recomputed == signed_output["output_hash"], "HASH_MISMATCH"

    # 3. Verify signature:
    pub_key = AGENT_PUBLIC_KEYS[expected_agent_id]
    key = RSA.import_key(pub_key)
    h   = SHA256.new(recomputed.encode())
    sig = base64.b64decode(signed_output["signature"])
    pkcs1_15.new(key).verify(h, sig)    ← raises ValueError if invalid

    return True   ← signature valid, content unmodified

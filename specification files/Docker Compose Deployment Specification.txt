3. nginx.conf (Key Routing)
text
upstream gateway {
    server api_gateway:8000;
    server api_gateway:8000;  # second replica
}

upstream frontend {
    server frontend:3000;
    server frontend:3000;
}

server {
    listen 443 ssl http2;
    server_name api.aiventra.internal;

    ssl_certificate     /etc/nginx/certs/aiventra.crt;
    ssl_certificate_key /etc/nginx/certs/aiventra.key;

    # REST API
    location /api/ {
        proxy_pass         http://gateway/;
        proxy_set_header   Host $host;
        proxy_set_header   X-Real-IP $remote_addr;
        proxy_read_timeout 120s;
        client_max_body_size 200M;  # large file uploads
    }

    # WebSocket upgrade
    location /ws/ {
        proxy_pass         http://gateway/ws/;
        proxy_http_version 1.1;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   Connection "upgrade";
        proxy_read_timeout 3600s;   # keep WS alive
    }

    # Frontend
    location / {
        proxy_pass http://frontend/;
    }
}

def test_henssge_estimate_midrange():
    result = henssge_estimate(
        temp_rectal=30.0,
        temp_ambient=18.0,
        weight_kg=70,
        correction_factor=1.0
    )
    # Known reference: ~10h PMI at these values
    assert 8.0 <= result.mean_hours <= 13.0
    assert result.lower_95 < result.mean_hours < result.upper_95

def test_henssge_rejects_impossible_temps():
    with pytest.raises(ValueError):
        henssge_estimate(temp_rectal=39.0, temp_ambient=18.0,
                         weight_kg=70, correction_factor=1.0)
        # Body temp above normal → can't estimate cooling

Layer 3: End-to-End Tests
python
# tests/e2e/test_full_murder_case.py

async def test_complete_murder_case_pipeline():
    """
    Uses the synthetic CASE-001 fixture:
    - CDR: 3 numbers, 48h of records
    - Autopsy PDF: blunt force trauma, fixed livor
    - Financial: 2 large cash withdrawals day before
    Expected outcome: HOMICIDE primary hypothesis > 0.70
    """
    case_id = await create_and_upload_case_001()
    run_id  = await trigger_pipeline(case_id)

    report  = await wait_for_pipeline_complete(run_id, timeout=300)

    # Core assertions:
    assert report["hypothesis"]["primary_hypothesis"] == "HOMICIDE"
    assert report["hypothesis"]["primary_confidence"] >= 0.70

    # Replay completeness:
    replay = report["reasoning_replay"]
    assert replay["total_steps"] >= 20
    assert len(replay["reasoning_gaps"]) == 0
    assert replay["overall_case_confidence"] >= 0.65

    # TOD consistency:
    tod = report["tod"]
    assert tod["consistency"]["rigor"] == "CONSISTENT"
    assert tod["consistency"]["livor"] == "CONSISTENT"

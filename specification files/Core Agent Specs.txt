def preprocess_image(image_bytes: bytes) -> bytes:
    import cv2, numpy as np
    from PIL import Image

    img      = cv2.imdecode(np.frombuffer(image_bytes, np.uint8), cv2.IMREAD_COLOR)
    gray     = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    denoised = cv2.fastNlMeansDenoising(gray, h=10, templateWindowSize=7, searchWindowSize=21)

    # Deskew:
    coords = np.column_stack(np.where(denoised < 128))
    if len(coords) > 0:
        angle = cv2.minAreaRect(coords)[-1]
        angle = -(90 + angle) if angle < -45 else -angle
        if abs(angle) > 0.5:
            h, w = denoised.shape[:2]
            M = cv2.getRotationMatrix2D((w//2, h//2), angle, 1.0)
            denoised = cv2.warpAffine(denoised, M, (w, h),
                           flags=cv2.INTER_CUBIC,
                           borderMode=cv2.BORDER_REPLICATE)

    binary = cv2.adaptiveThreshold(
        denoised, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
        cv2.THRESH_BINARY, 11, 2
    )
    _, buffer = cv2.imencode(".png", binary)
    return buffer.tobytes()

def detect_cdr_operator(df):
    col_set = set(df.columns.str.strip())
    scores  = {}
    for operator, sig in CDR_OPERATOR_SIGNATURES.items():
        req = len(sig["required_cols"] & col_set) / len(sig["required_cols"])
        opt = len(sig.get("optional_cols",set()) & col_set) / max(len(sig.get("optional_cols",set())),1)
        scores[operator] = req * 0.7 + opt * 0.3
    best = max(scores, key=scores.get)
    return (best, scores[best]) if scores[best] >= 0.5 else ("UNKNOWN", scores[best])

def normalize_msisdn(raw: str) -> str:
    digits = re.sub(r"\D", "", str(raw))
    if len(digits) == 10:             return f"91{digits}"
    if len(digits) == 11 and digits.startswith("0"):  return f"91{digits[1:]}"
    if len(digits) == 12 and digits.startswith("91"): return digits
    if len(digits) == 13 and digits.startswith("091"):return digits[1:]
    return digits

TIME_FORMATS = ["%H:%M:%S", "%H:%M", "%I:%M:%S %p", "%I:%M %p"]

def normalize_datetime(date_str, time_str, timezone="Asia/Kolkata") -> datetime:
    import pytz
    combined = f"{date_str.strip()} {time_str.strip()}"
    for dfmt in DATE_FORMATS:
        for tfmt in TIME_FORMATS:
            try:
                local_dt = datetime.strptime(combined, f"{dfmt} {tfmt}")
                tz       = pytz.timezone(timezone)
                return tz.localize(local_dt).astimezone(pytz.utc)
            except ValueError:
                continue
    # Fallback:
    from dateutil import parser as p
    local_dt = p.parse(combined, dayfirst=True)
    tz = pytz.timezone(timezone)
    if not local_dt.tzinfo:
        local_dt = tz.localize(local_dt)
    return local_dt.astimezone(pytz.utc)

Silence Window Detection
python
def find_silence_windows(events, min_silence_hours=4.0):
    sorted_ev = sorted(events, key=lambda e: e.event_timestamp)
    windows   = []
    for i in range(1, len(sorted_ev)):
        gap_h = (sorted_ev[i].event_timestamp -
                 sorted_ev[i-1].event_timestamp).total_seconds() / 3600
        if gap_h >= min_silence_hours:
            windows.append(SilenceWindow(
                start      = sorted_ev[i-1].event_timestamp,
                end        = sorted_ev[i].event_timestamp,
                duration_h = round(gap_h, 2)
            ))
    return windows
TOD Behavioral Delta
python
def compute_tod_behavioral_delta(events, tod_start, tod_end, baseline_days=7):
    tod_events = [e for e in events if tod_start <= e.event_timestamp <= tod_end]
    tod_hours  = {e.event_timestamp.hour for e in tod_events}
    cutoff     = tod_start - timedelta(days=baseline_days)
    baseline   = [e for e in events
                  if cutoff <= e.event_timestamp < tod_start
                  and e.event_timestamp.hour in tod_hours]
    daily_avg  = len(baseline) / baseline_days

    interpretation = (
        "SILENCE_DURING_TOD"         if len(tod_events)==0 and daily_avg>1 else
        "ELEVATED_ACTIVITY_DURING_TOD" if len(tod_events)>daily_avg*2 else
        "NORMAL"
    )
    return TodBehaviorDelta(
        tod_event_count    = len(tod_events),
        baseline_daily_avg = round(daily_avg, 2),
        ratio              = round(len(tod_events)/daily_avg,2) if daily_avg>0 else None,
        interpretation     = interpretation
    )
Tower Sequence Analysis

def build_tower_sequence(events):
    tower_events = sorted([e for e in events if e.cell_tower_id],
                          key=lambda e: e.event_timestamp)
    transitions  = []
    for i in range(1, len(tower_events)):
        prev, curr = tower_events[i-1], tower_events[i]
        if prev.cell_tower_id != curr.cell_tower_id:
            time_min = (curr.event_timestamp - prev.event_timestamp).total_seconds()/60
            speed    = None
            if prev.tower_latitude and curr.tower_latitude:
                dist_km = haversine(prev.tower_latitude, prev.tower_longitude,
                                    curr.tower_latitude, curr.tower_longitude)
                speed   = (dist_km/time_min*60) if time_min > 0 else None
            transitions.append(TowerTransition(
                from_tower    = prev.cell_tower_id,
                to_tower      = curr.cell_tower_id,
                time_diff_min = time_min,
                speed_kmh     = speed,
                anomalous     = speed is not None and speed > 200
            ))
    return TowerSequence(transitions=transitions,
                         anomalous_jumps=[t for t in transitions if t.anomalous])

def classify_transaction(narration: str) -> str:
    narration_upper = narration.upper()
    for category, patterns in TRANSACTION_CATEGORIES.items():
        if any(re.search(p, narration_upper) for p in patterns):
            return category
    return "UNKNOWN"

def compute_tod_financial_delta(events, tod_start, tod_end, baseline_days=30):
    tod_txns   = [e for e in events if tod_start <= e.timestamp <= tod_end]
    tod_debit  = sum(e.amount for e in tod_txns if e.txn_type=="DEBIT")
    tod_credit = sum(e.amount for e in tod_txns if e.txn_type=="CREDIT")

    cutoff     = tod_start - timedelta(days=baseline_days)
    baseline   = [e for e in events if cutoff <= e.timestamp < tod_start]
    daily_avg_debit = (sum(e.amount for e in baseline if e.txn_type=="DEBIT") / baseline_days
                       if baseline_days > 0 else 0)

    return TodFinancialDelta(
        tod_debit_total       = tod_debit,
        tod_credit_total      = tod_credit,
        baseline_daily_debit  = round(daily_avg_debit, 2),
        debit_ratio           = round(tod_debit / daily_avg_debit, 2) if daily_avg_debit else None,
        unusual_large_txn_in_window = any(t.amount > 100000 for t in tod_txns)
    )

# Sliding window: 1-hour buckets across case window
def extract_temporal_features(events, window_hours=1) -> pd.DataFrame:
    buckets = {}
    for ev in events:
        bucket_key = ev.timestamp.replace(minute=0, second=0, microsecond=0)
        if bucket_key not in buckets:
            buckets[bucket_key] = {
                "cdr_events":          0,
                "cdr_calls":           0,
                "cdr_sms":             0,
                "financial_txns":      0,
                "financial_debit_sum": 0,
                "device_events":       0,
                "unique_towers":       set(),
                "unique_contacts":     set()
            }
        b = buckets[bucket_key]
        if ev.stream_type == "CDR":
            b["cdr_events"] += 1
            if ev.event_type in ("MOC","MTC"):   b["cdr_calls"] += 1
            if ev.event_type in ("SMS_MO","SMS_MT"): b["cdr_sms"] += 1
            if ev.payload.cell_tower_id:         b["unique_towers"].add(ev.payload.cell_tower_id)
            if ev.payload.counterparty_msisdn:   b["unique_contacts"].add(ev.payload.counterparty_msisdn)
        elif ev.stream_type == "FINANCIAL":
            b["financial_txns"] += 1
            if ev.payload.txn_type == "DEBIT":   b["financial_debit_sum"] += ev.payload.amount
        elif ev.stream_type == "DEVICE":
            b["device_events"] += 1

    rows = []
    for ts, b in sorted(buckets.items()):
        rows.append({
            "bucket_ts":        ts,
            "cdr_events":       b["cdr_events"],
            "cdr_calls":        b["cdr_calls"],
            "cdr_sms":          b["cdr_sms"],
            "financial_txns":   b["financial_txns"],
            "financial_debit":  b["financial_debit_sum"],
            "device_events":    b["device_events"],
            "unique_towers":    len(b["unique_towers"]),
            "unique_contacts":  len(b["unique_contacts"])
        })
    return pd.DataFrame(rows)
Anomaly Detection (Two Models)

# Model 1: Isolation Forest (fast, interpretable)
from sklearn.ensemble import IsolationForest

def run_isolation_forest(df: pd.DataFrame) -> pd.Series:
    features = df.drop(columns=["bucket_ts"])
    model    = IsolationForest(contamination=0.05, random_state=42)
    scores   = model.fit_predict(features)
    # -1 = anomaly, 1 = normal
    return pd.Series(scores == -1, index=df.index, name="if_anomaly")

# Model 2: Autoencoder (captures temporal structure)
def build_autoencoder(input_dim: int):
    from tensorflow import keras
    encoder = keras.Sequential([
        keras.layers.Dense(16, activation="relu", input_shape=(input_dim,)),
        keras.layers.Dense(8,  activation="relu"),
        keras.layers.Dense(4,  activation="relu")
    ])
    decoder = keras.Sequential([
        keras.layers.Dense(8,  activation="relu", input_shape=(4,)),
        keras.layers.Dense(16, activation="relu"),
        keras.layers.Dense(input_dim, activation="linear")
    ])
    autoencoder = keras.Sequential([encoder, decoder])
    autoencoder.compile(optimizer="adam", loss="mse")
    return autoencoder

def run_autoencoder_anomaly(df: pd.DataFrame, threshold_percentile=95) -> pd.Series:
    features = df.drop(columns=["bucket_ts"]).values.astype(float)
    scaler   = StandardScaler()
    X        = scaler.fit_transform(features)

    ae = build_autoencoder(X.shape[1])
    ae.fit(X, X, epochs=50, batch_size=16, verbose=0)

    X_reconstructed = ae.predict(X)
    reconstruction_error = np.mean(np.square(X - X_reconstructed), axis=1)

    threshold = np.percentile(reconstruction_error, threshold_percentile)
    return pd.Series(reconstruction_error > threshold, name="ae_anomaly")

# Fusion: anomaly if BOTH models agree (reduces false positives)
def fuse_anomaly_scores(if_anomaly, ae_anomaly) -> pd.Series:
    return (if_anomaly & ae_anomaly).rename("fused_anomaly")

def build_anomaly_windows(df: pd.DataFrame,
                           fused_anomaly: pd.Series) -> List[AnomalyWindow]:
    """
    Merge consecutive anomalous buckets into windows.
    Tag each window with which streams contributed.
    """
    windows = []
    in_window = False
    for idx, row in df.iterrows():
        is_anomaly = fused_anomaly[idx]
        if is_anomaly and not in_window:
            window_start = row["bucket_ts"]
            in_window = True
        elif not is_anomaly and in_window:
            windows.append(AnomalyWindow(
                start           = window_start,
                end             = row["bucket_ts"],
                contributing_streams = detect_contributing_streams(df, window_start, row["bucket_ts"]),
                severity        = compute_window_severity(df, window_start, row["bucket_ts"])
            ))
            in_window = False

    return windows

M-09 — Hotspot / Correlation Engine
Purpose
The Hotspot Engine finds the intersection of: anomaly windows (from Timeline Agent) × suspicious contacts (from CDR Agent) × suspicious transactions (from Financial Agent) × geographic clusters (from tower data). Each intersection becomes a scored Hotspot that feeds the Hypothesis Manager.

Hotspot Types
Type	What it represents
TEMPORAL_HOTSPOT	High anomaly density in a time window
CONTACT_HOTSPOT	Unknown/suspicious contact cluster near TOD
FINANCIAL_HOTSPOT	Large/unusual transaction near TOD
GEOGRAPHIC_HOTSPOT	Tower location cluster near crime scene
COMPOUND_HOTSPOT	2+ hotspot types overlapping in same window
Intersection Algorithm
python
def compute_hotspots(
    anomaly_windows:     List[AnomalyWindow],
    cdr_result:          CdrAnalysisResult,
    financial_result:    FinancialAnalysisResult,
    tod_estimate:        TodResult
) -> List[Hotspot]:

    hotspots = []

    for window in anomaly_windows:
        contributing_factors = []
        score = 0.0

        # 1. TOD overlap:
        if window.tod_overlap:
            score += 0.3
            contributing_factors.append("TOD_WINDOW_OVERLAP")

        # 2. Unknown contacts in window:
        unknown_in_window = [c for c in cdr_result.unknown_contacts_near_tod
                             if window.start <= c.last_event <= window.end]
        if unknown_in_window:
            score += 0.2 * len(unknown_in_window)
            contributing_factors.append(f"UNKNOWN_CONTACTS:{len(unknown_in_window)}")

        # 3. Financial anomalies in window:
        fin_in_window = [a for a in financial_result.anomalies
                         if window.start <= a.timestamp <= window.end]
        if fin_in_window:
            score += 0.15 * len(fin_in_window)
            contributing_factors.append(f"FINANCIAL_ANOMALIES:{len(fin_in_window)}")

        # 4. Silence during window:
        silence_in_window = [s for s in cdr_result.silence_windows
                             if s.start >= window.start and s.end <= window.end]
        if silence_in_window:
            score += 0.15
            contributing_factors.append("CDR_SILENCE_IN_WINDOW")

        # 5. Anomalous tower jump in window:
        tower_jumps = [t for t in cdr_result.tower_sequence.anomalous_jumps
                       if window.start <= t.to_time <= window.end]
        if tower_jumps:
            score += 0.10
            contributing_factors.append("ANOMALOUS_TOWER_JUMP")

        if score > 0.2:    # threshold: only materialize significant hotspots
            hotspots.append(Hotspot(
                window_start         = window.start,
                window_end           = window.end,
                hotspot_type         = classify_hotspot_type(contributing_factors),
                composite_score      = min(round(score, 4), 1.0),
                contributing_factors = contributing_factors,
                severity             = score_to_severity(score)
            ))

    # Sort by score descending:
    hotspots.sort(key=lambda h: h.composite_score, reverse=True)
    return hotspots

HYPOTHESES = ["HOMICIDE", "SUICIDE", "ACCIDENT", "NATURAL", "UNDETERMINED"]

    "defensive_wounds_present": {

    "manner_of_death_homicide": {

    },

        "HOMICIDE":-0.20, "SUICIDE":+0.40, "ACCIDENT":-0.10, "NATURAL":-0.10
    },
    "lethal_toxicology": {
        "HOMICIDE":+0.10, "SUICIDE":+0.20, "ACCIDENT":+0.05, "NATURAL":+0.05
    },
    "silence_during_tod": {
        "HOMICIDE":+0.15, "SUICIDE":+0.05, "ACCIDENT":+0.05, "NATURAL":0.0
    },
    "unknown_contact_near_tod": {
        "HOMICIDE":+0.20, "SUICIDE":0.0, "ACCIDENT":+0.05, "NATURAL":0.0

    },
    "critical_hotspot": {
        "HOMICIDE":+0.20, "SUICIDE":+0.05, "ACCIDENT":+0.05, "NATURAL":0.0
    },
    "no_injuries": {
        "HOMICIDE":-0.20, "SUICIDE":-0.05, "ACCIDENT":-0.10, "NATURAL":+0.15
    },
    "natural_cause_of_death": {
        "HOMICIDE":-0.25, "SUICIDE":-0.15, "ACCIDENT":-0.15, "NATURAL":+0.35
    }

def score_hypotheses(
    autopsy:     AutopsyResult,
    cdr:         CdrAnalysisResult,
    financial:   FinancialAnalysisResult,
    hotspots:    HotspotResult,
    tod:         TodResult
) -> Dict[str, float]:

    scores = {h: 0.25 for h in HYPOTHESES}   # uniform prior

    # Map evidence → adjustment:
    evidence_signals = {
        "defensive_wounds_present":   autopsy.defensive_wounds == True,
        "manner_of_death_homicide":   autopsy.manner_of_death == "HOMICIDE",
        "manner_of_death_suicide":    autopsy.manner_of_death == "SUICIDE",
        "lethal_toxicology":          any(d.significance=="LETHAL" for d in autopsy.toxicology.drug_details),
        "silence_during_tod":         "SILENCE_WINDOW_DURING_TOD" in cdr.analysis_flags,
        "unknown_contact_near_tod":   bool(cdr.unknown_contacts_near_tod),
        "large_financial_txn_near_tod": "LARGE_TRANSACTION_NEAR_TOD" in financial.forensic_flags,
        "critical_hotspot":           any(h.severity=="CRITICAL" for h in hotspots.hotspots),
        "no_injuries":                not autopsy.injuries_present,
        "natural_cause_of_death":     autopsy.manner_of_death == "NATURAL"
    }

    for signal, is_present in evidence_signals.items():
        if is_present and signal in EVIDENCE_WEIGHTS:
            for hyp, delta in EVIDENCE_WEIGHTS[signal].items():
                scores[hyp] = max(0.0, scores[hyp] + delta)

    # Normalize to sum=1:
    total = sum(scores.values())
    return {h: round(s/total, 4) for h, s in scores.items()}
Contradiction Detection

def detect_contradictions(autopsy, cdr, financial, tod) -> List[str]:
    contradictions = []

    # Pathologist says SUICIDE but defensive wounds present:
    if autopsy.manner_of_death=="SUICIDE" and autopsy.defensive_wounds:
        contradictions.append("PATHOLOGIST_MANNER_VS_DEFENSIVE_WOUNDS")

    # Pathologist says NATURAL but high-severity hotspot near TOD:
    if autopsy.manner_of_death=="NATURAL" and any(h.severity in ("HIGH","CRITICAL")
                                                   for h in financial.anomalies):
        contradictions.append("NATURAL_MANNER_VS_FINANCIAL_ANOMALY")

    # TOD window entirely in silence period — no phone activity at all:
    if cdr.tod_behavior_delta.interpretation=="SILENCE_DURING_TOD":
        contradictions.append("COMPLETE_COMMUNICATION_BLACKOUT_DURING_TOD")

    # Unknown contact appeared immediately after TOD window:
    if cdr.unknown_contacts_near_tod:
        for c in cdr.unknown_contacts_near_tod:
            if c.first_event >= tod.todWindow95["start"]:
                contradictions.append("NEW_CONTACT_POST_TOD")

    return contradictions

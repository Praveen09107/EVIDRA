AIVENTRA — Layer 4: Presentation, Replay & Audit Layer
Full Technical Specification
Layer: 4 of 4 | Domain: Report Viewer, Reasoning Replay Viewer, Timeline Explorer, Hypothesis Dashboard, Audit Trail, WebSocket Client, Export Engine
Runtime: Next.js 15 (App Router) + React 19 | Port: 3000

PART I — LAYER PURPOSE & BOUNDARIES
Layer 4 is the human intelligence interface of AIVENTRA. It takes every structured output from Layer 3 and renders it into a coherent, interactive, audit-ready investigation workspace. It never writes to the database directly — all mutations go through Layer 1's API. All reads are authenticated REST calls that Layer 1 proxies.

Six core responsibilities:

Forensic Report Viewer — structured rendering of all 9 report sections with collapsible panels, confidence indicators, and inline evidence links

Reasoning Replay Viewer (RRV) — step-by-step playback of every inference decision with cryptographic chain verification badges, agent timeline, and hypothesis trajectory chart

Timeline Explorer — zoomable, filterable event timeline with anomaly overlay and hotspot highlighting

Hypothesis Dashboard — live-updating Bayesian posterior chart, evidence signal table, and trajectory animation

Audit Trail Viewer — immutable audit log per case with actor, action, and timestamp

Export Engine — PDF export of the full forensic report with evidence integrity certificates

Boundary contracts:

Reads from: Layer 1 REST API (/api/v1/*) — all authenticated, RBAC-enforced

Real-time: Layer 1 WebSocket (/ws/cases/{case_id}) for live pipeline updates

Writes to: Layer 1 REST API only (case notes, status updates, export requests)

Never: direct DB access, direct MinIO access, direct agent calls

// hooks/useWebSocket.ts

const WS_RECONNECT_DELAYS = [1000, 2000, 4000, 8000, 16000, 30000]; // ms

export function useWebSocket(caseId: string) {
  const ws = useRef<WebSocket | null>(null);
  const reconnectAttempt = useRef(0);
  const reconnectTimer = useRef<NodeJS.Timeout>();
  const [connected, setConnected] = useState(false);
  const [reconnecting, setReconnecting] = useState(false);
  const { dispatch } = usePipelineStore();
  const { accessToken } = useAuth();

  const connect = useCallback(() => {
    // Attach JWT as query param (WS doesn't support Authorization header):
    const url = `${WS_BASE_URL}/ws/cases/${caseId}?token=${accessToken}`;
    ws.current = new WebSocket(url);

    ws.current.onopen = () => {
      setConnected(true);
      setReconnecting(false);
      reconnectAttempt.current = 0;
      caseStore.setState({ wsConnected: true, wsReconnecting: false });
      // Send ping every 25s to keep connection alive:
      startHeartbeat();
    };

    ws.current.onmessage = (event) => {
      try {
        const msg: WsEvent = JSON.parse(event.data);
        handleWsEvent(msg);
      } catch (e) {
        console.error("WS parse error:", e);
      }
    };

    ws.current.onclose = (event) => {
      setConnected(false);
      stopHeartbeat();
      if (!event.wasClean) {
        scheduleReconnect();
      }
    };

    ws.current.onerror = () => {
      ws.current?.close();
    };
  }, [caseId, accessToken]);

  const scheduleReconnect = useCallback(() => {
    const delay = WS_RECONNECT_DELAYS[
      Math.min(reconnectAttempt.current, WS_RECONNECT_DELAYS.length - 1)
    ];
    reconnectAttempt.current += 1;
    setReconnecting(true);
    caseStore.setState({ wsReconnecting: true });
    reconnectTimer.current = setTimeout(connect, delay);
  }, [connect]);

  function handleWsEvent(event: WsEvent) {
    switch (event.event_type) {
      case "PIPELINE_STARTED":
        pipelineStore.setState({ status: "RUNNING", pipelineRunId: event.payload.pipeline_run_id });
        break;
      case "AGENT_STARTED":
        pipelineStore.setState(s => ({
          agentStatuses: { ...s.agentStatuses, [event.payload.agent_id]: "RUNNING" }
        }));
        break;
      case "AGENT_COMPLETE":
        pipelineStore.setState(s => ({
          agentStatuses:  { ...s.agentStatuses, [event.payload.agent_id]: "DONE" },
          agentDurations: { ...s.agentDurations, [event.payload.agent_id]: event.payload.duration_ms },
          totalTokens:    s.totalTokens + (event.payload.llm_tokens || 0)
        }));
        break;
      case "AGENT_ERROR":
        pipelineStore.setState(s => ({
          agentStatuses: { ...s.agentStatuses, [event.payload.agent_id]: "FAILED" }
        }));
        toast.error(`Agent ${event.payload.agent_id} failed: ${event.payload.error_summary}`);
        break;
      case "AGENT_RETRYING":
        pipelineStore.setState(s => ({
          agentStatuses: { ...s.agentStatuses, [event.payload.agent_id]: "RETRYING" }
        }));
        break;
      case "TOD_UPDATED":
        pipelineStore.setState({ todEstimate: event.payload });
        break;
      case "HYPOTHESIS_UPDATED":
        pipelineStore.setState({ latestHypothesis: event.payload });
        break;
      case "ANOMALY_FLAGGED":
        toast.info(`${event.payload.anomalous_event_count} anomalous events detected`);
        break;
      case "HOTSPOT_IDENTIFIED":
        toast.warning(`${event.payload.critical_count} critical hotspot windows identified`);
        break;
      case "REPORT_READY":
        pipelineStore.setState({
          status:       "DONE",
          reportReady:  true,
          reportId:     event.payload.report_id
        });
        toast.success("Forensic report ready");
        break;
      case "PIPELINE_COMPLETE":
        pipelineStore.setState({
          status:          "DONE",
          totalDurationMs: event.payload.total_duration_ms,
          totalTokens:     event.payload.total_llm_tokens_used
        });
        break;
      case "PIPELINE_FAILED":
        pipelineStore.setState({ status: "FAILED" });
        toast.error(`Pipeline failed: ${event.payload.error}`);
        break;
    }
  }

  useEffect(() => { connect(); return () => { ws.current?.close(); stopHeartbeat(); }; }, [caseId]);

  return { connected, reconnecting };
}
PART IV — PIPELINE STATUS BAR
The PipelineStatusBar renders at the top of every case page when a pipeline is running or recently completed. It shows the 11-agent DAG execution state in real-time.

typescript
// components/case/PipelineStatusBar.tsx

const AGENT_TIERS = {
  1: ["evidence_parser", "ocr", "format_normalizer"],
  2: ["autopsy", "tod", "digital_timeline", "anomaly_detector"],
  3: ["hotspot_correlation", "hypothesis_manager", "reasoning_replay", "report_generator"]
};

const AGENT_LABELS: Record<string, string> = {
  evidence_parser:     "Evidence Parser",
  ocr:                 "OCR",
  format_normalizer:   "Normalizer",
  autopsy:             "Autopsy",
  tod:                 "TOD Estimator",
  digital_timeline:    "Digital Timeline",
  anomaly_detector:    "Anomaly Detector",
  hotspot_correlation: "Hotspot Correlator",
  hypothesis_manager:  "Hypothesis Engine",
  reasoning_replay:    "Replay Chain",
  report_generator:    "Report Generator"
};

export function PipelineStatusBar() {
  const { status, agentStatuses, agentDurations, totalTokens, totalDurationMs }
    = usePipelineStore();

  if (status === "IDLE") return null;

  const completedCount = Object.values(agentStatuses).filter(s => s === "DONE").length;
  const totalAgents    = Object.keys(agentStatuses).length;
  const overallPct     = Math.round((completedCount / totalAgents) * 100);

  return (
    <div className="pipeline-status-bar" data-status={status}>
      {/* Overall progress strip */}
      <div className="pipeline-progress-track">
        <div className="pipeline-progress-fill" style={{ width: `${overallPct}%` }} />
      </div>

      {/* Tier swimlanes */}
      <div className="pipeline-tiers">
        {[1, 2, 3].map(tier => (
          <div key={tier} className="pipeline-tier">
            <span className="tier-label">Tier {tier}</span>
            <div className="tier-agents">
              {AGENT_TIERS[tier].map(agentId => (
                <AgentStatusChip
                  key={agentId}
                  agentId={agentId}
                  label={AGENT_LABELS[agentId]}
                  status={agentStatuses[agentId] ?? "PENDING"}
                  durationMs={agentDurations[agentId]}
                />
              ))}
            </div>
          </div>
        ))}
      </div>

      {/* Stats row */}
      {status === "DONE" && (
        <div className="pipeline-stats">
          <span>{completedCount}/{totalAgents} agents completed</span>
          <span>{(totalDurationMs / 1000).toFixed(1)}s total</span>
          <span>{totalTokens.toLocaleString()} LLM tokens</span>
        </div>
      )}

      {/* Live badge if running */}
      {status === "RUNNING" && <LiveBadge />}
    </div>
  );
}

// AgentStatusChip renders per-agent state:
export function AgentStatusChip({ agentId, label, status, durationMs }) {
  const COLOR_MAP = {
    PENDING:  "--color-text-faint",
    RUNNING:  "--color-primary",
    RETRYING: "--color-warning",
    DONE:     "--color-success",
    FAILED:   "--color-error",
    SKIPPED:  "--color-text-faint"
  };

  return (
    <div className={`agent-chip agent-chip--${status.toLowerCase()}`}
         title={durationMs ? `${durationMs}ms` : status}>
      <div className="agent-chip-dot"
           style={{ backgroundColor: `var(${COLOR_MAP[status]})` }}>
        {status === "RUNNING" && <div className="agent-chip-pulse" />}
      </div>
      <span className="agent-chip-label">{label}</span>
      {status === "DONE" && durationMs && (
        <span className="agent-chip-duration">{(durationMs/1000).toFixed(1)}s</span>
      )}
    </div>
  );
}

// TodBand: renders the visual time-of-death window
export function TodBand({ pointEstimate, lower95, upper95, lower68, upper68 }) {
  const totalSpanMs = new Date(upper95).getTime() - new Date(lower95).getTime();
  const pctOf = (t: string) =>
    ((new Date(t).getTime() - new Date(lower95).getTime()) / totalSpanMs) * 100;

  return (
    <div className="tod-band" role="img" aria-label="Time of death window visualization">
      {/* 95% CI band */}
      <div className="tod-band-track">
        <div className="tod-band-95"
             style={{ left: "0%", width: "100%" }}
             title={`95% CI: ${formatTodWindow(lower95, upper95)}`} />
        {/* 68% CI band */}
        {lower68 && upper68 && (
          <div className="tod-band-68"
               style={{
                 left:  `${pctOf(lower68)}%`,
                 width: `${pctOf(upper68) - pctOf(lower68)}%`
               }}
               title={`68% CI: ${formatTodWindow(lower68, upper68)}`} />
        )}
        {/* Point estimate needle */}
        <div className="tod-band-needle"
             style={{ left: `${pctOf(pointEstimate)}%` }}
             title={`Estimate: ${formatTod(pointEstimate)}`} />
      </div>

      {/* Axis labels */}

6.2 Client-Side Chain Verifier

// lib/crypto/verifyChain.ts

import { sha256 } from "js-sha256";   // tiny browser-compatible SHA-256

export interface ChainVerificationResult {
  valid:        boolean;
  totalSteps:   number;
  verifiedSteps:number;
  brokenAtSeq:  number | null;
  error:        string | null;
}

export function verifyReplayChain(replayData: ReplayChainResponse): ChainVerificationResult {
  const { steps, genesis_hash } = replayData;
  const sorted = [...steps].sort((a, b) => a.seq - b.seq);

  let prevHash = genesis_hash;
  let brokenAtSeq: number | null = null;

  for (const step of sorted) {
    // 1. Reconstruct canonical content (must match server logic exactly):
    const stepContent = JSON.stringify({
      seq:            step.seq,
      agent_id:       step.agent_id,
      trigger_type:   step.trigger_type,
      input_summary:  step.input_summary,
      interpretation: step.interpretation,
      confidence:     step.confidence !== null ? String(step.confidence) : "null",
      timestamp:      step.timestamp
    }, null);    // sort_keys is implicit since we define key order here

    // 2. Compute hash:
    const computedHash = sha256(stepContent);

    // 3. Verify step hash matches:
    if (computedHash !== step.step_hash) {
      brokenAtSeq = step.seq;
      return {
        valid:          false,
        totalSteps:     steps.length,
        verifiedSteps:  step.seq,
        brokenAtSeq,
        error:          `Step ${step.seq} hash mismatch (content tampered)`
      };
    }

    // 4. Verify chain linkage:
    if (step.prev_step_hash !== prevHash) {
      brokenAtSeq = step.seq;
      return {
        valid:          false,
        totalSteps:     steps.length,
        verifiedSteps:  step.seq,
        brokenAtSeq,
        error:          `Step ${step.seq} chain broken (prev_hash mismatch)`
      };
    }

    // Note: RSA signature verification is skipped client-side
    // (private key never leaves server; server pre-verified).
    // We verify content integrity + chain linkage client-side.

    prevHash = step.step_hash;
  }

  return {
    valid:          true,
    totalSteps:     steps.length,
    verifiedSteps:  steps.length,
    brokenAtSeq:    null,
    error:          null
  };
}

// components/replay/ReplayControls.tsx

export function ReplayControls({ totalSteps }) {
  const { currentSeq, isPlaying, playbackSpeed, setSeq, setPlaying, setSpeed }
    = useReplayStore();

  // Auto-advance when playing:
  useEffect(() => {
    if (!isPlaying) return;
    const interval = setInterval(() => {
      setSeq(s => {
        if (s >= totalSteps - 1) { setPlaying(false); return s; }
        return s + 1;
      });
    }, 1000 / playbackSpeed);
    return () => clearInterval(interval);
  }, [isPlaying, playbackSpeed, totalSteps]);

  return (
    <div className="replay-controls">
      {/* Step slider */}
      <input
        type    = "range"
        min     = {0}
        max     = {totalSteps - 1}
        value   = {currentSeq}
        onChange= {e => setSeq(Number(e.target.value))}
        className="replay-slider"
        aria-label="Replay position"
      />

      {/* Playback buttons */}
      <div className="replay-buttons">
        <button onClick={() => setSeq(0)}           aria-label="Go to start">⏮</button>
        <button onClick={() => setSeq(s => Math.max(0, s - 1))}  aria-label="Step back">⏪</button>

        <button
          onClick   = {() => setPlaying(!isPlaying)}
          className = {`replay-play-btn ${isPlaying ? "replay-play-btn--playing" : ""}`}
          aria-label= {isPlaying ? "Pause" : "Play"}
        >
          {isPlaying ? "⏸" : "▶"}
        </button>

        <button onClick={() => setSeq(s => Math.min(totalSteps - 1, s + 1))} aria-label="Step forward">⏩</button>
        <button onClick={() => setSeq(totalSteps - 1)} aria-label="Go to end">⏭</button>
      </div>

      {/* Speed selector */}
      <div className="replay-speed">
        <span>Speed:</span>

            key       = {s}
            onClick   = {() => setSpeed(s)}

        ))}

// components/timeline/TimelineCanvas.tsx

export function TimelineCanvas({ events, zoom, onZoomChange, overlays, onHover }) {
  const svgRef = useRef<SVGSVGElement>(null);
  const [width, setWidth] = useState(1200);
  const HEIGHT = 240;
  const AXIS_H  = 40;
  const TRACK_H = 180;
  const Y_TRACKS: Record<string, number> = {
    CDR:                 40,
    LOCATION_DATA:       90,
    FINANCIAL_RECORDS:  140,
  };

  // Compute visible time range:
  const allTimes = events.map(e => new Date(e.event_time).getTime());
  const minMs = zoom.startMs || Math.min(...allTimes);
  const maxMs = zoom.endMs   || Math.max(...allTimes);
  const spanMs = maxMs - minMs;

  const pctX = (ms: number) => ((ms - minMs) / spanMs) * width;

  // Resize observer:
  useEffect(() => {
    const obs = new ResizeObserver(entries => {
      setWidth(entries[0].contentRect.width);
    });
    if (svgRef.current?.parentElement) obs.observe(svgRef.current.parentElement);
    return () => obs.disconnect();
  }, []);

  // Zoom via wheel:
  const handleWheel = (e: WheelEvent) => {
    e.preventDefault();
    const zoomFactor = e.deltaY > 0 ? 1.2 : 0.85;
    const pivotPct   = e.offsetX / width;
    const pivotMs    = minMs + pivotPct * spanMs;
    const newSpan    = spanMs * zoomFactor;
    onZoomChange({
      startMs: pivotMs - pivotPct * newSpan,
      endMs:   pivotMs + (1 - pivotPct) * newSpan
    });
  };

  return (
    <div className="timeline-canvas-wrap">
      <svg
        ref      = {svgRef}
        width    = {width}
        height   = {HEIGHT}
        className= "timeline-svg"
        onWheel  = {handleWheel}
        role     = "img"
        aria-label="Forensic event timeline"
      >
        {/* TOD window overlay */}
        {overlays?.todWindow && (() => {
          const x1 = pctX(new Date(overlays.todWindow.lower95).getTime());
          const x2 = pctX(new Date(overlays.todWindow.upper95).getTime());
          return (
            <g className="tod-overlay">
              <rect x={x1} y={0} width={x2-x1} height={TRACK_H}
                    fill="var(--color-error)" opacity={0.06}
                    rx={4} />
              <line x1={(x1+x2)/2} y1={0} x2={(x1+x2)/2} y2={TRACK_H}
                    stroke="var(--color-error)" strokeWidth={1.5}
                    strokeDasharray="4 4" opacity={0.6} />
              <text x={(x1+x2)/2} y={14} textAnchor="middle"
                    fontSize={10} fill="var(--color-error)" opacity={0.8}>
                TOD window
              </text>
            </g>
          );
        })()}

        {/* Hotspot bands */}
        {overlays?.hotspots.map((h, i) => {
          const x1 = pctX(new Date(h.start).getTime());
          const x2 = pctX(new Date(h.end).getTime());
          const col = h.priority === "CRITICAL"
                        ? "var(--color-error)"
                        : h.priority === "HIGH"
                        ? "var(--color-warning)"
                        : "var(--color-gold)";
          return (
            <rect key={i} x={x1} y={0} width={Math.max(x2-x1, 3)}
                  height={TRACK_H} fill={col} opacity={0.08}
                  rx={2} />
          );
        })}

        {/* Track lanes */}
        {Object.entries(Y_TRACKS).map(([source, y]) => (
          <g key={source}>
            <line x1={0} y1={y} x2={width} y2={y}
                  stroke="var(--color-divider)" strokeWidth={1} />
            <text x={4} y={y - 6} fontSize={10}
                  fill="var(--color-text-faint)">
              {source.replace("_", " ")}
            </text>
          </g>
        ))}

        {/* Events */}
        {events.map(event => {
          const cx = pctX(new Date(event.event_time).getTime());
          const cy = Y_TRACKS[event.source] ?? 100;
          if (cx < -10 || cx > width + 10) return null; // out of view

          return (
            <EventDot
              key       = {event.id}
              cx        = {cx}
              cy        = {cy}
              event     = {event}
              onHover   = {onHover}
            />
          );
        })}

        {/* Time axis */}
        <TimeAxis minMs={minMs} maxMs={maxMs} width={width} y={TRACK_H + 20} />
      </svg>

      {/* Pan control strip */}
      <div className="timeline-pan-strip">
        <button onClick={() => onZoomChange({ startMs: 0, endMs: 0 })}>Reset</button>
        <input type="range" min={0} max={100}
               onChange={e => {/* pan handler */}} />
      </div>
    </div>
  );
}

export function EventDot({ cx, cy, event, onHover }) {
  const COLOR_MAP = {
    CALL:                   "var(--color-primary)",
    SMS:                    "var(--color-blue)",
    LOCATION_FIX:           "var(--color-success)",
    FINANCIAL_TRANSACTION:  "var(--color-gold)"
  };
  const r = event.is_anomalous ? 7 : 4;
  const fill = COLOR_MAP[event.event_type] ?? "var(--color-text-muted)";

  return (
    <g
      onMouseEnter = {() => onHover(event)}
      onMouseLeave = {() => onHover(null)}
      style        = {{ cursor: "pointer" }}
    >
      {event.is_anomalous && (
        <circle cx={cx} cy={cy} r={r + 4}
                fill="none"
                stroke={fill}
                strokeWidth={1.5}
                opacity={0.4} />
      )}
      <circle cx={cx} cy={cy} r={r}
              fill={fill}
              opacity={event.is_anomalous ? 1.0 : 0.75} />
    </g>
  );
}

// lib/export/pdfExport.ts

export async function triggerReportExport(reportId: string, caseId: string): Promise<void> {
  // 1. Request export from Layer 1 — triggers server-side PDF generation:
  const { exportJobId } = await api.reports.requestExport(reportId, {
    format:   "PDF",
    sections: "ALL",
    includeEvidence: false   // never include raw evidence bytes in export
  });

  // 2. Poll for completion (server uses WeasyPrint/Playwright to render PDF):
  toast.info("Generating PDF report…");
  let attempts = 0;
  while (attempts < 30) {
    await sleep(2000);
    const status = await api.reports.getExportStatus(exportJobId);
    if (status.status === "DONE") {
      // 3. Trigger browser download:
      const url = status.downloadUrl;    // pre-signed temporary URL (60s TTL)
      const link = document.createElement("a");
      link.href     = url;
      link.download = `AIVENTRA_Report_${caseId}_${reportId.slice(0, 8)}.pdf`;
      link.click();
      toast.success("Report downloaded");
      return;
    }
    if (status.status === "FAILED") {
      toast.error("PDF generation failed");
      return;
    }
    attempts++;
  }
  toast.error("PDF generation timed out");
}

# Export Worker:
async def _run_export_job(fields: dict):
    report_id = UUID(fields["report_id"])
    report    = await db.fetchrow("SELECT * FROM reports WHERE id=$1", report_id)
    report_data = json.loads(report["report_json"])

    # Render HTML template with report data:
    html_content = render_report_template(report_data)

    # Convert to PDF using Playwright (headless Chromium):
    async with async_playwright() as p:
        browser = await p.chromium.launch()
        page    = await browser.new_page()
        await page.set_content(html_content)
        await page.emulate_media(media="print")
        pdf_bytes = await page.pdf(
            format            = "A4",
            print_background  = True,
            margin            = {"top": "20mm", "bottom": "20mm",
                                  "left": "15mm", "right": "15mm"}
        )
        await browser.close()

    # Upload PDF to MinIO (temporary bucket, 1h expiry):
    pdf_key = f"exports/{report_id}/{uuid4()}.pdf"
    await minio.put_object("aiventra-exports", pdf_key, pdf_bytes,
                             content_type="application/pdf")

    # Generate pre-signed download URL (60s TTL):
    download_url = await minio.presigned_get_object(
        "aiventra-exports", pdf_key, expires=timedelta(seconds=60)
    )

    # Update export job status:
    await redis.hset(f"export_job:{export_job_id}",
                     mapping={"status": "DONE", "download_url": download_url})
